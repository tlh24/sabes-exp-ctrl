782  void makeStars(int nstars)
783  {
784  //distribute the stars uniformly over w, h.
785  //this just requires scaling some rand() s.
786  if (m_v) free(m_v);
787  m_v = NULL;
788  if (m_pvel) free(m_pvel);
789  m_pvel = NULL;
790  m_v = (starStruct *)malloc(nstars * sizeof(starStruct));
791  m_pvel = (float *)malloc(nstars * 2 * sizeof(float));
792  m_phase = (double *)malloc(nstars * sizeof(double));
793  for (int i=0; i<nstars; i++) {
794 
795  // draw uniformly in a circle with radius 0.5
796  float x = uniform()*PI*2.f;
797  float u = random()+random();
798  float r = u > 1 ? 2-u : u;
799  m_v[i].position[0] = 0.5f*r*cosf(x);
800  m_v[i].position[1] = 0.5f*r*sinf(x);
801 
802  m_v[i].color = 0xffffffff;
803  float angle = uniform()*PI*2.f;
804  m_pvel[i*2+0] = sinf(angle); // unit vector
805  m_pvel[i*2+1] = cosf(angle); // not a velocity
806  m_phase[i] = uniform()*PI*2.0;
807  }
808  m_n = nstars;
809  m_needConfig[0] = m_needConfig[1] = true;
810  m_drawmode = GL_POINTS;
811  }
812  virtual void move(long double time)
813  {
814 
815  if (!m_v || !m_pvel || !m_phase) return;
816 
817  unsigned int basecol = 0;
818  basecol += (unsigned int)(m_color[3] * 255) & 255;
819  basecol <<= 8;
820  basecol += (unsigned int)(m_color[2] * 255) & 255;
821  basecol <<= 8;
822  basecol += (unsigned int)(m_color[1] * 255) & 255;
823  basecol <<= 8;
824  basecol += (unsigned int)(m_color[0] * 255) & 255;
825 
826  float dt = (float)(time - m_lastTime);
827  double fq = 1/(m_lifetime+1e-6); // dot lifetime freq; protect div-0
828 
829  float scaled_vel[2];
830  scaled_vel[0] = m_vel[0] / m_scale[0];
831  scaled_vel[1] = m_vel[1] / m_scale[1];
832  float speed = norm(scaled_vel[0], scaled_vel[1]);
833 
834  int k = (int)floor(m_n * m_coherence);
835  for (int i=0; i<m_n; i++) {
836 
837  float p[2];
838  for (int j=0; j<2; j++) {
839  p[j] = m_v[i].position[j];
840  float v;
841  if (i<k) // coherent stars
842  v = scaled_vel[j];
843  else // random stars
844  v = m_pvel[i*2+j] * speed;
845  p[j] += v * dt;
846  }
847  float r = norm(m_v[i].position[0], m_v[i].position[1]);
848  if (r > 0.5 && speed > 0) {
849  float angle = atan2(p[1],p[0]);
850  angle += PI;
851  p[0] = 0.5 * cosf(angle);
852  p[1] = 0.5 * sinf(angle);
853  }
854  m_v[i].position[0] = p[0];
855  m_v[i].position[1] = p[1];
856 
857  m_v[i].color = basecol;
858 
859  double d = cos(2.0*PI*fq*time + m_phase[i]);
860  double dp = cos(2.0*PI*fq*m_lastTime + m_phase[i]); // previous
861 
862  // update the position.
863  // if lifetime is negative, stars live forever
864  if (d > 0 && dp <= 0 && m_lifetime > 0.f) {
865  // draw uniformly in a circle with radius 0.5
866  float a = uniform();
867  float b = uniform();
868  if (b < a) {
869  float tmp = b;
870  b = a;
871  a = tmp;
872  }
873  m_v[i].position[0] = 0.5f*b*cosf(PI*2.f*a/b);
874  m_v[i].position[1] = 0.5f*b*sinf(PI*2.f*a/b);
875  }
876  }
877  m_lastTime = time;
878  }
